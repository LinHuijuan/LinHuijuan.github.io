# 一、算法实现

  • 思路：
    • 设定特殊拐点
    • 缩短环数
<ul id="ul1"></ul>
<script>
    var oUl = document.getElementById('ul1');
    var aLi = oUl.getElementsByTagName('li');
    var size = 8;
    var len = size * size;
    var sizeGird = 50;
    var row = 0;
    var col = 0;
    var min = 0;
    var max = size - 1;
    var arr = [];

    oUl.style.width = size * (sizeGird + 1) + 'px';

    for (var i = 0; i < len; i++) {
        var oLi = document.createElement('li');
        oLi.style.width = sizeGird + 'px';
        oLi.style.height = sizeGird + 'px';
        oLi.style.listStyle = 'none';
        oLi.style.float = 'left';
        oUl.appendChild(oLi);
    }

    for (var i = 0; i < len; i++) {
        aLi[row * size + col].innerHTML = i;
        // aLi[row*size+col]相当于下标为[row.col]的li
        arr.push(aLi[row * size + col]);
        // 先朝右边横着过去
        if (row == min && col < max) {
            col = col + 1;
        }
        // 再竖着下来
        else if (col == max && row < max) {
            row = row + 1;
        }
        // 再朝左边横着过来
        else if (row == max && col > min) {
            col = col - 1;
        }
        // 再竖着上来
        else if (col == min && row > min) {
            row = row - 1;
        }
        // 缩小环数
        if (row - 1 == min && col == min) {
            min = min + 1;
            max = max - 1;
        }
    }

# 3.1.2 PageView

PageView 是一个可以完成页面之间滚动的widget。

  PageView({
    Key key,
    this.scrollDirection = Axis.horizontal,	// 滚动的方向，支持水平和垂直两个方向
    this.reverse = false,	// 是否反方向滚动
    PageController controller,	// PageView的控制类
    this.physics,	// 手势滑动逻辑，支持不滚动，总是滚动，与滚动边缘时是否有bounce
    this.pageSnapping = true,	// 设置为false以金庸页面捕捉，对自定义滚动行为很有用
    this.onPageChanged,	// 页面切换时调用
    List<Widget> children = const <Widget>[],
    this.dragStartBehavior = DragStartBehavior.start,
    this.allowImplicitScrolling = false,
  }) : assert(allowImplicitScrolling != null),
       controller = controller ?? _defaultPageController,
       childrenDelegate = SliverChildListDelegate(children),
       super(key: key);

# 3.1.3 项目实践

import 'package:flutter/material.dart';
import 'package:hrdesk_app/pages/homePage.dart';
import 'package:hrdesk_app/pages/myPage.dart';

class TabNavigator extends StatefulWidget {
  TabNavigator({Key key}) : super(key: key);
  @override
  _TabNavigatorState createState() => _TabNavigatorState();
}

// 类名加下划线，表示仅供内部访问
class _TabNavigatorState extends State<TabNavigator> {
  int _selectedIndex = 0; // 不能写在build里面，全局变量

  // final类型用于 使用过程中从来不会被修改的变量
  // initialPage: 0 初始状态下显示第一个tab
  final PageController _controller = PageController(initialPage: 0);

  void _onItemTapped(int index) {
    setState(() {
      _selectedIndex = index;
    });
    _controller.jumpToPage(index);
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: PageView(
        controller: _controller,
       	// children 存放要显示的页面
        children: <Widget>[HomePage(), MyPage()],
      ),
      bottomNavigationBar: BottomNavigationBar(
          currentIndex: _selectedIndex,
          selectedItemColor: Colors.blue,
          onTap: _onItemTapped,
          type: BottomNavigationBarType.fixed,
          items: [	// 必传参数，数组格式
            BottomNavigationBarItem(
              icon: Icon(Icons.home),
              title: Text('首页'),
            ),
            BottomNavigationBarItem(
              icon: Icon(Icons.account_circle),
              title: Text('我的'),
            ),
          ]),
    );
  }
}

child: NotificationListener(
    onNotification: (scrollNotification) {
        // scrollNotification is ScrollUpdateNotification 有滚动的时候才触发
        // scrollNotification.depth == 0 child的第0个widget，这里即ListView列表
        // 避免未滚动时也触发了监听函数，或者轮播图滚动时触发了函数
        if (scrollNotification is ScrollUpdateNotification &&
            scrollNotification.depth == 0) {
            //滚动且是列表滚动的时候
            _onScroll(scrollNotification.metrics.pixels);
        }
        return;
    },
    child: ListView()
)

double appBarAlpha = 0;

_onScroll(offset) {
    double alpha = offset / APPBAR_SCROLL_OFFSET;
    if (alpha < 0) {
        alpha = 0;
    } else if (alpha > 1) {
        alpha = 1;
    }
    setState(() {
        appBarAlpha = alpha;
    });
}

# 6.2 滚动字母表触发列表滚动

// Alphabet.vue

<li
    class="item"
    v-for="item of letters"
    :key="item"
    :ref="item"
    @touchstart.prevent="handleTouchStart"
    @touchmove="handleTouchMove"
    @touchend="handleTouchEnd"
    @click="handleLetterClick"
    >
    {{item}}
</li>

<script>
export default {
  name: 'CityAlphabet',
  props: {
    cities: Object
  },
  // 用数组储存字母，方便通过下标获取数组
  computed: {
    letters () {
      const letters = []
      for (let i in this.cities) {
        letters.push(i)
      }
      return letters
    }
  },
  data () {
    return {
      // 标识是否触摸字母表
      touchStatus: false,
      startY: 0,
      timer: null
    }
  },
  // 7.1 性能优化
  // 由于数据更改导致的虚拟 DOM 重新渲染和打补丁，在这之后会调用该钩子。
  updated () {
    // 距离列表顶部的长度
    this.startY = this.$refs['A'][0].offsetTop
  },
  methods: {
    handleLetterClick (e) {
      this.$emit('change', e.target.innerText)
    },
    handleTouchStart () {
      // 开始触摸
      this.touchStatus = true
    },
    handleTouchMove (e) {
      // 触摸时执行
      if (this.touchStatus) {
        // 7.2 性能优化
        if (this.timer) {
          clearTimeout(this.timer)
        }
        this.timer = setTimeout(() => {
          // e.touches[0]表示的是手指的一些信息
          // 距离整个页面顶部的长度减去头部的高度
          const touchY = e.touches[0].clientY - 79
          // 获取字母的下标，因为是从0开始，所以向下取整
          const index = Math.floor((touchY - this.startY) / 20)
          // 在字母表区域才触发
          if (index >= 0 && index < this.letters.length) {
            this.$emit('change', this.letters[index])
          }
        }, 16)
      }
    },
    handleTouchEnd () {
      // 结束触摸
      this.touchStatus = false
    }
  }
}
</script>

# 八、搜索功能实现

// Search.vue
<template>
  <div>
    <div class="search">
      <input v-model="keyword" class="search-input" type="text" placeholder="输入城市名或拼音" />
    </div>
    <div
      class="search-content"
      ref="search"
      v-show="keyword"
    >
      <ul>
        <li
          class="search-item border-bottom"
          v-for="item of list"
          :key="item.id"
        >
          {{item.name}}
        </li>
        <li class="search-item border-bottom" v-show="hasNoData">
          没有找到匹配数据
        </li>
      </ul>
    </div>
  </div>
</template>

<script>
import Bscroll from 'better-scroll'
export default {
  name: 'CitySearch',
  props: {
    cities: Object
  },
  data () {
    return {
      keyword: '',
      list: [],
      timer: null
    }
  },
  computed: {
    hasNoData () {
      return !this.list.length
    }
  },
  watch: {
    keyword () {
      if (this.timer) {
        clearTimeout(this.timer)
      }
      if (!this.keyword) {
        this.list = []
        return
      }
      this.timer = setTimeout(() => {
        const result = []
        for (let i in this.cities) {
          this.cities[i].forEach((value) => {
            if (value.spell.indexOf(this.keyword) > -1 || value.name.indexOf(this.keyword) > -1) {
              result.push(value)
            }
          })
        }
        this.list = result
      }, 100)
    }
  },
  mounted () {
    this.scroll = new Bscroll(this.$refs.search)
  }
}
</script>

# 9.1 Vuex

Vuex 相当于一个store仓库，存储公用数据。

  • state：存放数据的地方。
  • action：将操作放在这里。
    • Action 提交的是 mutation，而不是直接变更状态。
    • Action 可以包含任意异步操作。
  • mutation：更改 Vuex 的 store 中的状态的唯一方法是提交 mutation。

	},
	...mapMutations(['changeCity'])

# 十一、使用keep-alive优化网页性能

# 11.1 存储路由内容

问题：在控制台点击Network下的XHR会发现，当每次路由发生切换的时候，Ajax都会重新被发送。

原因：每次路由切换时，组件都会被重新渲染，mouted钩子就会被重新执行，那么Ajax数据就会被重新获取。

解决：

// App.vue
<template>
  <div id="app">
    <keep-alive>
      <router-view/>
    </keep-alive>
  </div>
</template>

keep-alive 可以实现路由内容被加载一次后，就把路由内容放在内存之中，下一次进入路由时就无需重新渲染组件，直接从内存把内容拿出来显示即可。

# 11.2 切换城市时首页推荐也被切换

使用keep-alive时，组件里会执行一个activated生命周期函数。

activated在页面被重新显示时会执行。

// Home.vue
import { mapState } from 'vuex'
export default {
  name: 'Home',
  data () {
    return {
      lastCity: '',
      swiperList: [],
      iconList: [],
      recommendList: [],
      weekendList: []
    }
  },
  computed: {
    ...mapState(['city'])
  },
  methods: {
    getHomeInfo () {
	  // 返回带city参数的数据
      axios.get('/api/index.json?city=' + this.city)
        .then(this.getHomeInfoSucc)
    },
    getHomeInfoSucc (res) {
      res = res.data
      if (res.ret && res.data) {
        const data = res.data
        this.swiperList = data.swiperList
        this.iconList = data.iconList
        this.recommendList = data.recommendList
        this.weekendList = data.weekendList
      }
    }
  },
  mounted () {
    this.lastCity = this.city
    this.getHomeInfo()
  },
  activated () {
	// 当城市发生切换时，重新请求Ajax发送数据
    if (this.lastCity !== this.city) {
      this.lastCity = this.city
      this.getHomeInfo()
    }
  }
}
